# ⏰ Random time between 8 PM to 10 PM
start_time = datetime.now().replace(hour=20, minute=0, second=0, microsecond=0)
end_time = datetime.now().replace(hour=22, minute=0, second=0, microsecond=0)

# Pick random time in the window
random_time = start_time + timedelta(seconds=random.randint(0, int((end_time - start_time).total_seconds())))
wait_seconds = (random_time - datetime.now()).total_seconds()

# Wait until random time if not already past
if wait_seconds > 0:
    print(f"⏳ Waiting for {int(wait_seconds // 60)} minutes...")
    time.sleep(wait_seconds)
else:
    print("⚡ Time already in window, sending immediately.")

# 🎯 Pick random message set
selected_set = random.choice(messages)
body = "\n".join(selected_set + signature)

subject = "🌌 Good Night!"

# 📧 Send emails
try:
    server = smtplib.SMTP("smtp.gmail.com", 587)
    server.starttls()
    server.login(sender_email, password)

    for receiver in receivers:
        msg = MIMEText(body)
        msg["Subject"] = subject
        msg["From"] = sender_email
        msg["To"] = receiver

        server.sendmail(sender_email, receiver, msg.as_string())
        print(f"✅ Sent to {receiver}")

    server.quit()

except Exception as e:
    print("❌ Error:", e)
